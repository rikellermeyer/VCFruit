#make heatmap
#find frequency of values in bins of a given width 
num_bins=100   #desired number of bins in heatmap
chrm_len=int(134019835)  #define chromosome length in bp 
binwidth=chrm_len/int(num_bins)
binlist=list(np.linspace(0, chrm_len, num=num_bins, dtype=int)) #create list of the lower boundaries for each bin
hm_xax=np.linspace(binwidth, chrm_len, num=num_bins, dtype=int)#create the x-axis for the heatmap. Center of each bin in list
#make a dictionary that lists the bins as keys, frequency of variants within as value
counts_dict={}
for i in binlist:
    binmax=i+binwidth
    binmin=i
    for var in VCF_pos_list:
        if var<=binmax and var>binmin:
            if f'{binmin}-{binmax}' not in counts_dict:
                counts_dict[f'{binmin}-{binmax}']=1
            else:
                counts_dict[f'{binmin}-{binmax}']=counts_dict[f'{binmin}-{binmax}']+1

#make an array from the dictionary to allow heatmap generation
counts_list_by_bin=list(counts_dict.values()) #converts dictionary to list of counts 
#Heatmap requires a 2D array. To create a 2d data structure from a 1D list, replicate list x times to make a list of replicated lists, then convert to array
counts_list_by_bin_rep=[]  #empty list, will iterate in the replicated lists below
for i in range(10):        # number of replicates   
    counts_list_by_bin_rep.append(counts_list_by_bin)    
counts_array=np.array(counts_list_by_bin_rep) #creates an array with replicated counts list
fig, ax = plt.subplots(figsize=(50, 5))
counts_heatmap=sb.heatmap(counts_array, cmap='rainbow', yticklabels=False, xticklabels=hm_xax)

#First, need to make a dictionary that calculates the number of sequences that do NOT have a variant within the specified bin (span of chromosome)
novar_counts_dict={}
for bin in counts_dict:
    novar_counts_dict[bin]=(len(mock_whole)-counts_dict[bin])

#make dictionary of expected counts per bin assuming random distribution across chromosome
total_num_variants=len(mock_whole)
expected_var_per_bin=total_num_variants/num_bins
expected_dict={}
expected_novar_dict={}
for bin in counts_dict:
    expected_dict[bin]=expected_var_per_bin
    expected_novar_dict[bin]=total_num_variants-expected_var_per_bin

print('total vars:', total_num_variants)
#find difference between observed and expected count, represent as fold change 
enrichment_dict={}
for bin in counts_dict:
    enrichment_dict[bin]=counts_dict[bin]/expected_dict[bin]

#make dataframe with variant and no_variant counts in separate columns, bins=rows 
counts_df=pd.DataFrame([counts_dict, novar_counts_dict, expected_dict, expected_novar_dict, enrichment_dict])
